ex:

select irec_produto, convert_from(convert_to(irec_produto, 'UTF8'), 'LATIN1'),
       desc_produto,  convert_from(convert_to(desc_produto, 'UTF8'), 'LATIN1') from itemreceita
where irec_produto is not null or  desc_produto is not null       

select irec_recomendacao from itemreceita
where irec_produto is not null or  desc_produto is not null

set client_encoding = 'latin1';
